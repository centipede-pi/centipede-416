# This script sets up a competition between four people
#   The competitors gather in a circle, each wired to an eStim output
#   the system activates one output at a time, counting up
#   hitting contact 1 starts the count up over and advances to the next person

SET  1.1 ON=0         # power off everyone
SET  1.2 ON=0
SET  2.1 ON=0
SET  2.2 ON=0

SET  1.1 WAVE=BASIC
SET  1.2 WAVE=BASIC
SET  2.1 WAVE=BASIC
SET  2.2 WAVE=BASIC

SET  1.1 LEVEL=0%
SET  1.2 LEVEL=0%
SET  2.1 LEVEL=0%
SET  2.2 LEVEL=0%

PROMPT Set all sequence sliders to a hard limit value

#########################################################################

LABEL START1
SET  1.1 ON=1         # power off everyone
SET  1.2 ON=0
SET  2.1 ON=0
SET  2.2 ON=0
SET  1.1 LEVEL=0%
SET  1.2 LEVEL=0%
SET  2.1 LEVEL=0%
SET  2.2 LEVEL=0%
SPEECH STARTING TO RAISE THE LEVEL ON COMPETITOR ONE

SET  1.1 LEVEL=10%
SPEECH LEVEL ONE
WAIT SECONDS=10

SET  1.1 LEVEL=15%
SPEECH LEVEL TWO
WAIT SECONDS=10
